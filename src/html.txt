            const stockBody = document.getElementById('stockBody');

                stockBody.innerHTML = states.loading;
            } else if (state === 'content') {
                weatherBody.innerHTML = states.weatherContent;
                stockBody.innerHTML = states.stockContent;

                stockBody.innerHTML = states.error;
            }

            const isWeather = widgetId === 'weatherWidget';
